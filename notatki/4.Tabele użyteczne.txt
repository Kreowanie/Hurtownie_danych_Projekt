CREATE TABLE [HD_04].[dbo].[DimDate]
(
    DateKey INT PRIMARY KEY,
    FullDate DATE NOT NULL,
    DayNumber INT NOT NULL,
    MonthNumber INT NOT NULL,
    MonthName NVARCHAR(20) NOT NULL,
    QuarterNumber INT NOT NULL,
    YearNumber INT NOT NULL,
    IsWeekend BIT NOT NULL
);

INSERT INTO [HD_04].[dbo].[DimDate]
(
    DateKey,
    FullDate,
    DayNumber,
    MonthNumber,
    MonthName,
    QuarterNumber,
    YearNumber,
    IsWeekend
)
SELECT
    DateKey,
    CAST(MIN(TransactionDate) AS DATE) AS FullDate,
    DAY(MIN(TransactionDate)) AS DayNumber,
    MONTH(MIN(TransactionDate)) AS MonthNumber,
    DATENAME(MONTH, MIN(TransactionDate)) AS MonthName,
    DATEPART(QUARTER, MIN(TransactionDate)) AS QuarterNumber,
    YEAR(MIN(TransactionDate)) AS YearNumber,
    CAST(MAX(CAST(IsWeekend AS INT)) AS BIT) AS IsWeekend
FROM [HD_04].[dbo].[SalesClean]
WHERE DateKey IS NOT NULL
GROUP BY DateKey;


CREATE TABLE [HD_04].[dbo].[DimProduct]
(
    ProductKey INT IDENTITY(1,1) PRIMARY KEY,
    ProductNo VARCHAR(30) NULL,
    ProductName NVARCHAR(200) NULL,
    ProductCategory NVARCHAR(60) NULL
);

INSERT INTO [HD_04].[dbo].[DimProduct]
(
    ProductNo,
    ProductName,
    ProductCategory
)
SELECT DISTINCT
    ProductNo,
    ProductName,
    ProductCategory
FROM [HD_04].[dbo].[SalesClean]
WHERE ProductNo IS NOT NULL
   OR ProductName IS NOT NULL;


CREATE TABLE [HD_04].[dbo].[DimCustomer]
(
    CustomerKey INT IDENTITY(1,1) PRIMARY KEY,
    CustomerNo INT NULL,
    IsCustomerKnown BIT NOT NULL,
    CustomerType NVARCHAR(30) NOT NULL
);

INSERT INTO [HD_04].[dbo].[DimCustomer]
(
    CustomerNo,
    IsCustomerKnown,
    CustomerType
)
SELECT DISTINCT
    CustomerNo,
    IsCustomerKnown,
    CASE
        WHEN IsCustomerKnown = 1 THEN 'Znany klient'
        ELSE 'Brak klienta'
    END AS CustomerType
FROM [HD_04].[dbo].[SalesClean];

 CREATE TABLE [HD_04].[dbo].[DimCountry]
 (
     CountryKey INT IDENTITY(1,1) PRIMARY KEY,
     Country NVARCHAR(80) NULL,
     CountryGroup NVARCHAR(30) NULL,
     IsUnitedKingdom BIT NULL
 );
 GO

 WITH CleanCountry AS
 (
     SELECT DISTINCT
         LTRIM(RTRIM(
             CASE
                 WHEN Country LIKE '%,%'
                 THEN RIGHT(Country, CHARINDEX(',', REVERSE(Country)) - 1)
                 ELSE Country
             END
         )) AS Country
     FROM [HD_04].[dbo].[SalesClean]
     WHERE Country IS NOT NULL
 )
 INSERT INTO [HD_04].[dbo].[DimCountry]
 (
     Country,
     CountryGroup,
     IsUnitedKingdom
 )
 SELECT DISTINCT
     Country,
     CASE
         WHEN Country = N'United Kingdom' THEN N'United Kingdom'
         ELSE N'Export'
     END AS CountryGroup,
     CASE
         WHEN Country = N'United Kingdom' THEN 1
         ELSE 0
     END AS IsUnitedKingdom
 FROM CleanCountry
 WHERE Country IS NOT NULL
   AND Country <> N'';
 GO
